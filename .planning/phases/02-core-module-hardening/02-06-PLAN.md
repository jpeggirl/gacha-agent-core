```typescript
import { Bot } from 'grammy';
import { apiThrottler } from '@grammyjs/transformer-throttler';
import type { StorageAdapter, ScoredDeal, DealAlert, GachaAgentConfig } from '../types/index.js';

const SENT_ALERTS_STORAGE_KEY = 'alerts:sent-keys';

export class TelegramAlerts {
  private bot: Bot;
  private defaultChatId?: string;
  private sentAlerts: Set<string> = new Set();
  private storage: StorageAdapter;
  private dedupLoaded = false;

  constructor(config: GachaAgentConfig, storage: StorageAdapter) {
    if (!config.telegram?.botToken) {
      throw new Error('Telegram bot token is required');
    }
    this.bot = new Bot(config.telegram.botToken);
    this.bot.api.config.use(apiThrottler());
    this.defaultChatId = config.telegram.defaultChatId;
    this.storage = storage;
  }
```

2. **Lazy-load dedup keys from storage:**
```typescript
  private async ensureDedupLoaded(): Promise<void> {
    if (this.dedupLoaded) return;
    const stored = await this.storage.get<string[]>(SENT_ALERTS_STORAGE_KEY) ?? [];
    this.sentAlerts = new Set(stored);
    this.dedupLoaded = true;
  }
```

3. **Update sendDealAlert()** to load dedup and persist new keys:
```typescript
  async sendDealAlert(deal: ScoredDeal, watchlistEntryId: string, chatId?: string): Promise<DealAlert | null> {
    const targetChatId = chatId ?? this.defaultChatId;
    if (!targetChatId) throw new Error('No chat ID provided and no default chat ID configured');

    await this.ensureDedupLoaded();
    const dedupeKey = `${deal.listing.itemId}:${targetChatId}`;
    if (this.sentAlerts.has(dedupeKey)) return null;

    const message = this.formatDealMessage(deal);
    await this.bot.api.sendMessage(targetChatId, message, {
      parse_mode: 'HTML',
      link_preview_options: { is_disabled: false },
    });

    this.sentAlerts.add(dedupeKey);
    await this.storage.set(SENT_ALERTS_STORAGE_KEY, Array.from(this.sentAlerts));

    return {
      id: randomUUID(),
      deal,
      watchlistEntryId,
      sentAt: new Date().toISOString(),
      channel: 'telegram',
    };
  }
```

4. **Update sendText()** to use grammy:
```typescript
  async sendText(chatId: string, text: string): Promise<void> {
    await this.bot.api.sendMessage(chatId, text, { parse_mode: 'HTML' });
  }
```

5. **Keep** `formatDealMessage()` unchanged — it already produces correct HTML.

6. **Update clearDedupeCache()** to also clear storage:
```typescript
  async clearDedupeCache(): Promise<void> {
    this.sentAlerts.clear();
    this.dedupLoaded = false;
    await this.storage.delete(SENT_ALERTS_STORAGE_KEY);
  }
```
Note: `clearDedupeCache()` return type changes from `void` to `Promise<void>`.

class InMemoryStorage implements StorageAdapter {
  private data = new Map<string, unknown>();
  async get<T>(key: string): Promise<T | null> { return (this.data.get(key) as T) ?? null; }
  async set<T>(key: string, value: T): Promise<void> { this.data.set(key, value); }
